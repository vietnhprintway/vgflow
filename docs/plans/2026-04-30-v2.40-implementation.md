```python
#!/usr/bin/env python3
"""canonicalize_url.py — strip volatile params, normalize URL for memoization.

Usage: canonicalize_url.py <url>
Outputs canonical form to stdout.
"""
import sys
from urllib.parse import urlparse, urlunparse, parse_qsl, urlencode

VOLATILE_PARAMS = {
    "t", "_t", "ts", "timestamp", "session", "session_id", "sessionid",
    "request_id", "rid", "trace_id", "nonce", "_", "cb", "cachebust",
    "csrf", "_csrf", "auth_token", "token",
}

def canonicalize(url: str) -> str:
    p = urlparse(url)
    netloc = p.netloc.lower()
    path = p.path.rstrip("/") or "/"
    if path == "/":
        path = ""
    qs = [(k, v) for k, v in parse_qsl(p.query, keep_blank_values=True)
          if k.lower() not in VOLATILE_PARAMS]
    qs.sort()
    return urlunparse((p.scheme.lower(), netloc, path, "", urlencode(qs), ""))

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: canonicalize_url.py <url>", file=sys.stderr)
        sys.exit(2)
    print(canonicalize(sys.argv[1]))
```

```python
#!/usr/bin/env python3
"""identify-interesting-clickables.py — classify scan-*.json elements.

Reads scan-*.json, outputs recursive-classification.json with element_class
per clickable. Pure Python, deterministic, no LLM cost.

Output schema:
{
  "clickables": [
    {
      "view": "/admin/topup-requests",
      "element_class": "mutation_button",
      "selector": "button#delete-42",
      "selector_hash": "<sha256[:8]>",
      "resource": "topup_requests",
      "action_semantic": "delete",
      "metadata": {...}
    }
  ]
}
"""
from __future__ import annotations
import argparse, hashlib, json, re, sys
from pathlib import Path

MUTATION_METHODS = {"POST", "PUT", "PATCH", "DELETE"}
REDIRECT_PARAM_RE = re.compile(r"\b(redirect_uri|return_to|next|continue)\b", re.I)
URL_FETCH_PARAM_RE = re.compile(r"\b(url|link|webhook|callback|fetch_from)\b", re.I)
PATH_PARAM_RE = re.compile(r"\b(file|path|template|name)\b", re.I)
AUTH_ENDPOINT_RE = re.compile(r"^/api/auth/", re.I)

def selector_hash(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8")).hexdigest()[:8]

def classify_scan(scan: dict) -> list[dict]:
    out = []
    view = scan.get("view", "")

    # mutation_button: results with mutation network calls
    for r in scan.get("results", []) or []:
        for n in r.get("network", []) or []:
            if n.get("method", "").upper() in MUTATION_METHODS:
                sel = r.get("selector") or r.get("action", "")
                out.append({
                    "view": view, "element_class": "mutation_button",
                    "selector": sel, "selector_hash": selector_hash(sel),
                    "action_semantic": _action_from_method(n.get("method"), n.get("path", "")),
                    "metadata": {"method": n.get("method"), "path": n.get("path")},
                })

    # form_trigger
    for f in scan.get("forms", []) or []:
        if "submit_result" in f:
            sel = f.get("selector", "")
            ec = "file_upload" if any(fld.get("type") == "file" for fld in f.get("fields", []) or []) else "form_trigger"
            out.append({
                "view": view, "element_class": ec,
                "selector": sel, "selector_hash": selector_hash(sel),
                "action_semantic": "submit",
                "metadata": {"fields": f.get("fields", [])},
            })

    # tabs
    for t in scan.get("tabs", []) or []:
        out.append({
            "view": view, "element_class": "tab",
            "selector": f"tab[{t}]", "selector_hash": selector_hash(t),
            "action_semantic": "switch", "metadata": {"label": t},
        })

    # row_actions / bulk_actions
    for tbl in scan.get("tables", []) or []:
        for ra in tbl.get("row_actions", []) or []:
            out.append({
                "view": view, "element_class": "row_action",
                "selector": f"row_action[{ra}]", "selector_hash": selector_hash(ra),
                "action_semantic": ra, "metadata": {},
            })
        for ba in tbl.get("bulk_actions", []) or []:
            out.append({
                "view": view, "element_class": "bulk_action",
                "selector": f"bulk[{ba}]", "selector_hash": selector_hash(ba),
                "action_semantic": ba, "metadata": {},
            })

    # modal triggers
    for m in scan.get("modal_triggers", []) or []:
        out.append({
            "view": view, "element_class": "modal_trigger",
            "selector": m, "selector_hash": selector_hash(m),
            "action_semantic": "open_modal", "metadata": {},
        })

    # sub_view_link
    for sv in scan.get("sub_views_discovered", []) or []:
        out.append({
            "view": view, "element_class": "sub_view_link",
            "selector": f"link[{sv}]", "selector_hash": selector_hash(sv),
            "action_semantic": "navigate", "metadata": {"target": sv},
        })

    return out

def _action_from_method(method: str, path: str) -> str:
    m = (method or "").upper()
    if m == "DELETE": return "delete"
    if m == "POST": return "create"
    if m in ("PUT", "PATCH"): return "update"
    return "mutate"

def main() -> int:
    ap = argparse.ArgumentParser(description=__doc__)
    ap.add_argument("--scan-files", nargs="+", required=True, help="One or more scan-*.json")
    ap.add_argument("--output", default=None, help="Path to write recursive-classification.json")
    ap.add_argument("--json", action="store_true", help="Print result to stdout instead of writing file")
    args = ap.parse_args()

    all_clickables = []
    for sp in args.scan_files:
        p = Path(sp)
        if not p.is_file():
            print(f"⛔ scan file not found: {p}", file=sys.stderr)
            return 1
        scan = json.loads(p.read_text(encoding="utf-8"))
        all_clickables.extend(classify_scan(scan))

    payload = {"clickables": all_clickables, "count": len(all_clickables)}
    if args.json or args.output is None:
        print(json.dumps(payload, indent=2))
    if args.output:
        Path(args.output).write_text(json.dumps(payload, indent=2), encoding="utf-8")
    return 0

if __name__ == "__main__":
    sys.exit(main())
```

```python
#!/usr/bin/env python3
"""aggregate-recursive-goals.py — single-writer goal aggregator for Phase 2b-2.5.

Reads runs/goals-*.partial.yaml, dedupes via canonical key, writes:
- TEST-GOALS-DISCOVERED.md (capped per mode)
- recursive-goals-overflow.json (rest)

Canonical key: sha256(view|selector_hash|action_semantic|lens|resource|assertion_type)[:12]
"""
from __future__ import annotations
import argparse, hashlib, json, sys, yaml
from pathlib import Path

MODE_CAPS = {"light": 50, "deep": 150, "exhaustive": 400}

def canonical_key(g: dict) -> str:
    parts = [
        g.get("view", ""),
        g.get("selector_hash", "") or g.get("stable_selector", ""),
        g.get("action_semantic", ""),
        g.get("lens", ""),
        g.get("resource", ""),
        g.get("assertion_type", ""),
    ]
    return hashlib.sha256("|".join(parts).encode()).hexdigest()[:12]

def render_goal_md(g: dict) -> str:
    return f"""## G-RECURSE-{g['_canonical']}
- source: review.recursive_probe
- depth: {g.get('depth', 1)}
- lens: {g['lens']}
- view: {g['view']}
- element_class: {g.get('element_class')}
- selector_hash: {g.get('selector_hash')}
- resource: {g['resource']}
- parent_goal_id: {g.get('parent_goal_id', 'null')}
- priority: {g.get('priority', 'medium')}
"""

def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--phase-dir", required=True)
    ap.add_argument("--mode", choices=list(MODE_CAPS), default="light")
    ap.add_argument("--output", default=None, help="Path to TEST-GOALS-DISCOVERED.md (append-merge)")
    ap.add_argument("--overflow", default=None)
    args = ap.parse_args()

    phase_dir = Path(args.phase_dir).resolve()
    runs_dir = phase_dir / "runs"
    if not runs_dir.is_dir():
        print(f"⛔ runs/ missing: {runs_dir}", file=sys.stderr); return 1

    seen: dict[str, dict] = {}
    for partial in sorted(runs_dir.glob("goals-*.partial.yaml")):
        try:
            entries = yaml.safe_load(partial.read_text(encoding="utf-8")) or []
        except yaml.YAMLError as e:
            print(f"⚠ malformed {partial}: {e}", file=sys.stderr); continue
        for g in entries:
            k = canonical_key(g)
            g["_canonical"] = k
            if k not in seen:
                seen[k] = g

    cap = MODE_CAPS[args.mode]
    deduped = list(seen.values())
    main_goals = deduped[:cap]
    overflow_goals = deduped[cap:]

    output = Path(args.output) if args.output else (phase_dir / "TEST-GOALS-DISCOVERED.md")
    overflow_path = Path(args.overflow) if args.overflow else (phase_dir / "recursive-goals-overflow.json")

    existing = output.read_text(encoding="utf-8") if output.is_file() else ""
```

```python
#!/usr/bin/env python3
"""spawn-recursive-probe.py — Phase 2b-2.5 manager dispatcher.

Eligibility check (6 rules) → classify clickables → pick lenses per element class
→ spawn workers (auto) OR generate prompts (manual) → enforce 8 termination guards.
"""
from __future__ import annotations
import argparse, datetime as _dt, json, os, sys, yaml
from pathlib import Path

ELIGIBLE_PROFILES = {"feature", "feature-legacy", "hotfix"}
ELIGIBLE_SURFACES = {"ui", "ui-mobile"}

def check_eligibility(phase_dir: Path, override_reason: str | None) -> dict:
    if override_reason:
        return {"passed": False, "skipped_via_override": True, "reasons": [f"override: {override_reason}"]}
    reasons = []
    profile_file = phase_dir / ".phase-profile"
    profile_data = yaml.safe_load(profile_file.read_text()) if profile_file.is_file() else {}
    profile = profile_data.get("phase_profile", "")
    surface = profile_data.get("surface", "")
    if profile not in ELIGIBLE_PROFILES:
        reasons.append(f"phase_profile '{profile}' not in {ELIGIBLE_PROFILES}")
    if surface not in ELIGIBLE_SURFACES:
        reasons.append(f"surface '{surface}' not in {ELIGIBLE_SURFACES}")

    crud = phase_dir / "CRUD-SURFACES.md"
    if not crud.is_file() or "name:" not in crud.read_text():
        reasons.append("CRUD-SURFACES.md declares 0 resources")

    env = phase_dir / "ENV-CONTRACT.md"
    if not env.is_file():
        reasons.append("ENV-CONTRACT.md missing (rule 6)")
    else:
        env_text = env.read_text()
        if "disposable_seed_data: true" not in env_text:
            reasons.append("disposable_seed_data not true")
        # Check all third_party_stubs marked stubbed
        for line in env_text.split("\n"):
            if ":" in line and "third_party_stubs" in env_text and not line.strip().endswith("stubbed") and "stubbed" not in line and ":" in line.strip() and not line.strip().startswith("#"):
                pass  # simplified

    return {"passed": len(reasons) == 0, "reasons": reasons, "skipped_via_override": False}

def write_skip_evidence(phase_dir: Path, eligibility: dict) -> None:
    out = phase_dir / ".recursive-probe-skipped.yaml"
    out.write_text(yaml.safe_dump({
        "skipped_at": _dt.datetime.now(_dt.timezone.utc).isoformat(),
        "reasons": eligibility["reasons"],
        "via_override": eligibility.get("skipped_via_override", False),
    }), encoding="utf-8")

def main() -> int:
    ap = argparse.ArgumentParser(description=__doc__)
    ap.add_argument("--phase-dir", required=True)
    ap.add_argument("--mode", choices=["light", "deep", "exhaustive"], default="light")
    ap.add_argument("--probe-mode", choices=["auto", "manual", "hybrid"], default="auto")
    ap.add_argument("--skip-recursive-probe", default=None, help="Override reason; logs OVERRIDE-DEBT")
    ap.add_argument("--dry-run", action="store_true")
    ap.add_argument("--json", action="store_true")
```
